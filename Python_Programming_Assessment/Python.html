def create_non_overlapping_windows(
    chromosome_size: int,
    window_size: int,
    print_result: bool = False
):
    windows = []
    start_position = 1 #It is 1 , because the question defines it as 1

    while start_position <= chromosome_size:
        end_position = start_position + window_size - 1 #calculation of the intended end position

        #check if the end position is greater than chromosome size:
    
        if end_position > chromosome_size:
            end_position = chromosome_size

        # saving the current window:
        windows.append((start_position,end_position))
        # new start_position for the next window, it starts immediately after the current one ends
        start_position = end_position + 1

        #optional print argument:
        if print_result :
            print(f"Window: {start_position - window_size}-{end_position}") # as we updetaed the start_position in the former line
                                                                            # we should substract window_size to get the real start_positioon
    return windows

from collections import namedtuple 

SequenceInfo = namedtuple(
    'SequenceInfo',
    ['Length','A','T','C','G']
)

def sequence_info(dna_string):
    Length = len(dna_string)
    countA = dna_string.count("A")
    countT = dna_string.count("T")
    countC = dna_string.count("C")
    countG = dna_string.count("G")

    return SequenceInfo(Length = Length, A = countA, T = countT, C= countC, G =countG)

        "New_Sequence": new_sequence
